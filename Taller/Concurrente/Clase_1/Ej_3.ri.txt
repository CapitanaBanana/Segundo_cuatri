programa Ejercicio3
procesos
  proceso limpiarEsquina(ES cantFlores: numero; ES cantPapeles:numero)
  comenzar
    mientras (HayPapelEnLaEsquina)
      tomarPapel
      cantPapeles:= cantPapeles+1
    mientras(HayFlorEnLaEsquina)
      tomarFlor
      cantFlores:=cantFlores+1
    mientras (HayFlorEnLaBolsa)
      depositarFlor
    mientras (HayPapelEnLaBolsa)
      depositarPapel
  fin
  proceso escalon(E altura: numero; ES cumple:numero)
  variables
    cantPapeles:numero
    cantFlores:numero
  comenzar
    cantFlores:=0
    cantPapeles:=0
    repetir 2
      repetir (altura) 
        si (((PosCa)<100) & ((PosAv)<100))
          limpiarEsquina(cantFlores,cantPapeles) 
          mover
      derecha
    si (cantPapeles= ((cantFlores)+1))
      cumple:=(cumple)+1
  fin
  proceso escalerita
  variables
    cumple:numero
    altura:numero
  comenzar
    cumple:=0
    altura:=1
    repetir 13
      escalon(altura,cumple)
      altura:=altura+1
      repetir 2
        derecha
    Informar (cumple)
  fin
